# 第 13 课：混合架构实战 —— FastAPI 网关 + gRPC 微服务

> **目标**：实现一个真实项目中常见的混合架构：对外 REST (FastAPI)，对内 gRPC 微服务  
> **场景**：外面用 HTTP 服务浏览器/App，里面用 gRPC 做微服务间高性能通信

---

## 一、架构图示

```
┌─────────┐     HTTP/JSON      ┌──────────────┐     gRPC/Protobuf     ┌──────────────┐
│  浏览器  │ ─────────────────→ │  FastAPI 网关 │ ───────────────────→ │  用户微服务   │
│  / App   │ ←───────────────── │  (port 8000)  │ ←─────────────────── │  (port 50051) │
└─────────┘                     └──────────────┘                       └──────────────┘
                                      │                                        │
                                      │          gRPC/Protobuf                 │
                                      └───────────────────────────────────────┘
                                                        │
                                                 ┌──────┴──────┐
                                                 │  订单微服务   │
                                                 │  (port 50052) │
                                                 └──────────────┘
```

- **FastAPI 网关（port 8000）**：对外提供 REST 接口，处理 HTTP 请求
- **用户微服务（port 50051）**：对内 gRPC 服务，处理用户业务
- **订单微服务（port 50052）**：对内 gRPC 服务，处理订单业务

## 三、第一步：gRPC 用户微服务

和之前写的完全一样，独立运行：

```python
# grpc_user_server.py
import grpc
from concurrent import futures
import user_pb2
import user_pb2_grpc

class UserService(user_pb2_grpc.UserServiceServicer):
    def __init__(self):
        self.db = {}
        self.next_id = 1

    def CreateUser(self, request, context):
        user = user_pb2.UserResponse(
            id=self.next_id,
            name=request.name,
            email=request.email
        )
        self.db[self.next_id] = user
        self.next_id += 1
        print(f"[gRPC] 创建用户: {user.name} (ID={user.id})")
        return user

    def GetUser(self, request, context):
        if request.id not in self.db:
            context.abort(grpc.StatusCode.NOT_FOUND, f"User {request.id} not found")
        return self.db[request.id]

    def ListUsers(self, request, context):
        for user in self.db.values():
            yield user

def serve_grpc():
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    user_pb2_grpc.add_UserServiceServicer_to_server(UserService(), server)
    server.add_insecure_port('[::]:50051')
    server.start()
    print("[gRPC] gRPC 用户服务: [::]:50051")
    server.wait_for_termination()

if __name__ == '__main__':
    serve_grpc()
```

## 四、第二步：FastAPI 网关

网关接收 HTTP 请求，内部通过 gRPC Stub 调用微服务：

```python
# fastapi_gateway.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import grpc
import user_pb2
import user_pb2_grpc

app = FastAPI(title="用户服务 API 网关")

# ===== 初始化 gRPC 连接（应用启动时创建，全局复用）=====
GRPC_USER_SERVICE = "localhost:50051"

try:
    channel = grpc.insecure_channel(GRPC_USER_SERVICE)
    stub = user_pb2_grpc.UserServiceStub(channel)
    print(f"[OK] 已连接 gRPC 用户服务: {GRPC_USER_SERVICE}")
except Exception as e:
    print(f"[ERR] 无法连接 gRPC 服务: {e}")
    stub = None

# ===== Pydantic 模型（对外 REST 的 DTO）=====
class UserCreate(BaseModel):
    name: str
    email: str

class UserResponse(BaseModel):
    id: int
    name: str
    email: str

# ===== REST 接口（内部调用 gRPC）=====

@app.post("/api/users", response_model=UserResponse)
def create_user(user: UserCreate):
    """
    REST: POST /api/users
    内部: 调用 gRPC UserService.CreateUser
    """
    if stub is None:
        raise HTTPException(status_code=503, detail="gRPC 服务不可用")

    try:
        # 构造 gRPC 请求
        grpc_req = user_pb2.CreateUserRequest(
            name=user.name,
            email=user.email
        )
        # 调用 gRPC
        grpc_resp = stub.CreateUser(grpc_req, timeout=2.0)

        # 转成 REST 响应格式
        return UserResponse(
            id=grpc_resp.id,
            name=grpc_resp.name,
            email=grpc_resp.email
        )
    except grpc.RpcError as e:
        # 把 gRPC 错误映射为 HTTP 错误
        if e.code() == grpc.StatusCode.ALREADY_EXISTS:
            raise HTTPException(status_code=409, detail=e.details())
        elif e.code() == grpc.StatusCode.INVALID_ARGUMENT:
            raise HTTPException(status_code=400, detail=e.details())
        else:
            raise HTTPException(status_code=500, detail=f"Internal: {e.details()}")


@app.get("/api/users/{user_id}", response_model=UserResponse)
def get_user(user_id: int):
    """
    REST: GET /api/users/{id}
    内部: 调用 gRPC UserService.GetUser
    """
    if stub is None:
        raise HTTPException(status_code=503, detail="gRPC 服务不可用")

    try:
        grpc_req = user_pb2.GetUserRequest(id=user_id)
        grpc_resp = stub.GetUser(grpc_req, timeout=2.0)

        return UserResponse(
            id=grpc_resp.id,
            name=grpc_resp.name,
            email=grpc_resp.email
        )
    except grpc.RpcError as e:
        if e.code() == grpc.StatusCode.NOT_FOUND:
            raise HTTPException(status_code=404, detail=e.details())
        else:
            raise HTTPException(status_code=500, detail=f"Internal: {e.details()}")


@app.get("/api/users")
def list_users():
    """
    REST: GET /api/users
    内部: 调用 gRPC UserService.ListUsers（服务端流式）
    """
    if stub is None:
        raise HTTPException(status_code=503, detail="gRPC 服务不可用")

    try:
        # 服务端流式调用
        grpc_stream = stub.ListUsers(
            user_pb2.ListUsersRequest(),
            timeout=10.0
        )

        users = []
        for user in grpc_stream:
            users.append({
                "id": user.id,
                "name": user.name,
                "email": user.email
            })
        return users
    except grpc.RpcError as e:
        raise HTTPException(status_code=500, detail=e.details())


@app.get("/api/health")
def health_check():
    """健康检查：确认网关 + gRPC 都正常"""
    if stub is None:
        return {"status": "degraded", "grpc": "disconnected"}
    try:
        # 尝试获取一个已知用户
        stub.GetUser(user_pb2.GetUserRequest(id=1), timeout=1.0)
    except grpc.RpcError as e:
        if e.code() == grpc.StatusCode.NOT_FOUND:
            return {"status": "healthy", "grpc": "connected"}
        return {"status": "degraded", "grpc": f"error: {e.code()}"}
    return {"status": "healthy", "grpc": "connected"}


# ===== 启动 =====
if __name__ == '__main__':
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

## 五、第三步：一键启动

```python
# run_all.py
import threading
import uvicorn
import grpc_user_server

def start_grpc():
    """在子线程启动 gRPC 服务"""
    grpc_user_server.serve_grpc()

if __name__ == '__main__':
    # 启动 gRPC 微服务（后台线程）
    grpc_thread = threading.Thread(target=start_grpc, daemon=True)
    grpc_thread.start()

    # 等 gRPC 启动完成
    import time
    time.sleep(1)

    # 启动 FastAPI 网关（主线程）
    print("[HTTP] FastAPI 网关: http://localhost:8000")
    print("[BOOK] API 文档: http://localhost:8000/docs")
    uvicorn.run("fastapi_gateway:app", host="0.0.0.0", port=8000)
```

运行：

```bash
python run_all.py
```

```

[HTTP] FastAPI 网关: http://localhost:8000
[BOOK] API 文档: http://localhost:8000/docs
```

## 八、这种架构的好处

| 优点 | 说明 |
|------|------|
| **对前端友好** | 浏览器可以直接调 REST 接口，不需要 grpc-web |
| **对内高性能** | 微服务之间用 gRPC，Protobuf + HTTP/2 |
| **解耦** | FastAPI 网关可以随时换，gRPC 微服务不变 |
| **易调试** | 网关有 Swagger UI，微服务有 grpcurl |
| **易扩展** | 新增微服务只需在网关加对应 REST 接口 |
